82 template <class T>
83 PhysicalCudaMemory<T>* cudaPhysicalCalloc(size_t nelem, size_t gran) {
84  AvoidCudaGraphCaptureGuard cgcGuard;
85 
86  int deviceId = -1;
87  MSCCLPP_CUDATHROW(cudaGetDevice(&deviceId));
88 
89  CUmemAllocationProp prop = {};
90  prop.type = CU_MEM_ALLOCATION_TYPE_PINNED;
91  prop.location.type = CU_MEM_LOCATION_TYPE_DEVICE;
92  prop.location.id = deviceId;
93 #if defined(__HIP_PLATFORM_AMD__)
94  // TODO: revisit when HIP fixes this typo in the field name
95  prop.requestedHandleType = CU_MEM_HANDLE_TYPE_POSIX_FILE_DESCRIPTOR;
96 #else
97  prop.requestedHandleTypes = CU_MEM_HANDLE_TYPE_POSIX_FILE_DESCRIPTOR;
98 #endif
99 
100  CUmemGenericAllocationHandle memHandle;
101  size_t bufferSize = sizeof(T) * nelem;
102  // allocate physical memory
103  MSCCLPP_CUTHROW(cuMemCreate(&memHandle, bufferSize, &prop, 0 /*flags*/));
104 
105  CUmemAccessDesc accessDesc = {};
106  accessDesc.location.type = CU_MEM_LOCATION_TYPE_DEVICE;
107  accessDesc.location.id = deviceId;
108  accessDesc.flags = CU_MEM_ACCESS_FLAGS_PROT_READWRITE;
109 
110  T* devicePtr = nullptr;
111  // Map the device pointer
112  MSCCLPP_CUTHROW(cuMemAddressReserve((CUdeviceptr*)&devicePtr, bufferSize, gran, 0U, 0));
113  MSCCLPP_CUTHROW(cuMemMap((CUdeviceptr)devicePtr, bufferSize, 0, memHandle, 0));
114  MSCCLPP_CUTHROW(cuMemSetAccess((CUdeviceptr)devicePtr, bufferSize, &accessDesc, 1));
115  CudaStreamWithFlags stream(cudaStreamNonBlocking);
116  MSCCLPP_CUDATHROW(cudaMemsetAsync(devicePtr, 0, bufferSize, stream));
117 
118  MSCCLPP_CUDATHROW(cudaStreamSynchronize(stream));
119 
120  return new PhysicalCudaMemory<T>(memHandle, devicePtr, bufferSize);
121 }

160 template <class T, T*(alloc)(size_t), class Deleter, class Memory>
161 Memory safeAlloc(size_t nelem) {
162  T* ptr = nullptr;
163  try {
164  ptr = alloc(nelem);
165  } catch (...) {
166  if (ptr) {
167  Deleter()(ptr);
168  }
169  throw;
170  }
171  return Memory(ptr, Deleter());
172 }

174 template <class T, T*(alloc)(size_t, size_t), class Deleter, class Memory>
175 Memory safeAlloc(size_t nelem, size_t gran) {
176  if ((nelem * sizeof(T)) % gran) {
177  throw Error("The request allocation size is not divisible by the required granularity:" +
178  std::to_string(nelem * sizeof(T)) + " vs " + std::to_string(gran),
179  ErrorCode::InvalidUsage);
180  }
181  T* ptr = nullptr;
182  try {
183  ptr = alloc(nelem, gran);
184  } catch (...) {
185  if (ptr) {
186  Deleter()(ptr);
187  }
188  throw;
189  }
190  return Memory(ptr, Deleter());
191 }
