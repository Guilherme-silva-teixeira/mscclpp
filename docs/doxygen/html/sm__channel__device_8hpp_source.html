72  template <typename T, bool CopyRemainder = true>
73  MSCCLPP_DEVICE_INLINE void copy_helper(void* dst, void* src, uint64_t bytes, uint32_t threadId, uint32_t numThreads) {
74  int* dstInt = reinterpret_cast<int*>(dst);
75  int* srcInt = reinterpret_cast<int*>(src);
76  const uintptr_t dstPtr = reinterpret_cast<uintptr_t>(dst);
77  const uintptr_t srcPtr = reinterpret_cast<uintptr_t>(src);
78  const uint64_t numInt = bytes / sizeof(int);
79  T* dstElem = reinterpret_cast<T*>((dstPtr + sizeof(T) - 1) / sizeof(T) * sizeof(T));
80  T* srcElem = reinterpret_cast<T*>((srcPtr + sizeof(T) - 1) / sizeof(T) * sizeof(T));
81  uint64_t nFirstInt = (reinterpret_cast<uintptr_t>(dstElem) - dstPtr) / sizeof(int);
82  if (CopyRemainder) {
83  // Copy the remainder integers at the beginning.
84  Element::copy<int>(dstInt, srcInt, nFirstInt, threadId, numThreads);
85  }
86  // Copy elements.
87  constexpr uint64_t nIntPerElem = sizeof(T) / sizeof(int);
88  uint64_t nElem = (numInt - nFirstInt) / nIntPerElem;
89  Element::copy<T>(dstElem, srcElem, nElem, threadId, numThreads);
90  if (CopyRemainder && nIntPerElem > 1) {
91  // Copy the remainder integers at the end.
92  uint64_t nLastInt = (numInt - nFirstInt) % nIntPerElem;
93  Element::copy<int>(dstInt + nFirstInt + nElem * nIntPerElem, srcInt + nFirstInt + nElem * nIntPerElem, nLastInt,
94  threadId, numThreads);
95  }
96  }

111  template <int Alignment = 16, bool CopyRemainder = true>
112  MSCCLPP_DEVICE_INLINE void copy(void* dst, void* src, uint64_t bytes, uint32_t threadId, uint32_t numThreads) {
113  if (Alignment == 4) {
114  copy_helper<int, CopyRemainder>(dst, src, bytes, threadId, numThreads);
115  } else if (Alignment == 8) {
116  copy_helper<long long, CopyRemainder>(dst, src, bytes, threadId, numThreads);
117  } else if (Alignment == 16) {
118  copy_helper<longlong2, CopyRemainder>(dst, src, bytes, threadId, numThreads);
119  } else {
120  static_assert(Alignment == 4 || Alignment == 8 || Alignment == 16, "Unsupported alignment");
121  }
